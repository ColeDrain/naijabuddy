% Pandoc custom template — two-column IEEE/ACM-style academic paper.
% Targets the 4-8 page hackathon submission length cap. Title and
% abstract span both columns, body content flows in two columns, wide
% tables and figures use \begin{table*}/\begin{figure*} to span both.
% Typography stays Times (mathptmx) + microtype + booktabs for the
% expected academic look.

\documentclass[$if(fontsize)$$fontsize$$else$10pt$endif$,letterpaper,twocolumn]{article}

% Section/TOC depth — must be set BEFORE pandoc's body-block executes
% because pandoc emits $secnumdepth$ verbatim into a \setcounter call,
% and a missing value renders as the literal string "none" which crashes.
%
% secnumdepth=-1 disables LaTeX-generated section numbers. The markdown
% sources already author their own ("## 1. Introduction", "## 4.2 Rating
% Prediction"), and letting LaTeX number on top produces doubled prefixes
% like "0.1 1. Introduction". The numbering authored in the source is the
% one that matches the cross-references in §X.Y citations throughout the
% paper, so we keep the source numbers and silence LaTeX.
\setcounter{secnumdepth}{-1}
\setcounter{tocdepth}{2}

% ----- Encoding -------------------------------------------------------------
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{textcomp}

% ----- Typography (Times + microtype = "looks like a real paper") ----------
\usepackage{mathptmx}                 % Times for text + math
\usepackage[scaled=0.92]{helvet}      % Helvetica for sans-serif (sparingly)
\usepackage{courier}
\usepackage{microtype}                % Character protrusion + font expansion

% ----- Math -----------------------------------------------------------------
\usepackage{amsmath,amssymb,mathtools}

% ----- Layout / page geometry ----------------------------------------------
% Two-column IEEE-Transactions-style: 0.65in side margins, 0.85in top/
% bottom, modest column gap. With 10pt Times this lands ~55 chars per
% column, matching published IEEE/ACM dual-column papers. Tighter than
% the default for letter paper, but well within readability.
\usepackage[letterpaper,top=0.85in,bottom=0.85in,left=0.65in,right=0.65in,columnsep=0.3in]{geometry}
\linespread{1.0}                      % single spacing — two-column convention

% ----- Sectioning ----------------------------------------------------------
\usepackage{titlesec}
\titleformat{\section}{\large\bfseries}{\thesection}{0.6em}{}
\titleformat{\subsection}{\normalsize\bfseries}{\thesubsection}{0.6em}{}
\titleformat{\subsubsection}{\normalsize\itshape}{\thesubsubsection}{0.6em}{}
\titlespacing*{\section}{0pt}{1.4ex plus 0.3ex minus 0.2ex}{0.7ex plus 0.2ex}
\titlespacing*{\subsection}{0pt}{1.1ex plus 0.3ex minus 0.2ex}{0.5ex plus 0.2ex}

% ----- Lists ---------------------------------------------------------------
\usepackage{enumitem}
\setlist{nosep,leftmargin=2em}

% ----- Tables --------------------------------------------------------------
\usepackage{booktabs}
\usepackage{array}
\usepackage{longtable}
\usepackage{tabularx}

% NOTE: pandoc-emitted longtable blocks are rewritten to plain tabular
% by the post-processing step in scripts/build_pdfs.sh (longtable does
% not work inside a twocolumn document). See that script for details.

% ----- Floats & captions ---------------------------------------------------
\usepackage{float}
\usepackage[font=small,labelfont=bf,format=hang,justification=centering]{caption}

% ----- Hyperlinks ----------------------------------------------------------
\usepackage[colorlinks=true,linkcolor=blue,citecolor=blue,urlcolor=blue]{hyperref}
\usepackage{url}
\urlstyle{same}

% ----- Page style ----------------------------------------------------------
\pagestyle{plain}
% Tighter paragraph spacing for two-column layout — 0.3 baseline saves
% several lines across a multi-page paper vs the 0.5 we used in single-
% column. Indent first lines (academic convention) so we don't lose
% visual paragraph cues to the small parskip.
\setlength{\parskip}{0.3\baselineskip}
\setlength{\parindent}{0pt}

% ----- Unicode character mappings ------------------------------------------
% Greek letters and math symbols used inline outside math mode.
\DeclareUnicodeCharacter{03B1}{\ensuremath{\alpha}}
\DeclareUnicodeCharacter{03B2}{\ensuremath{\beta}}
\DeclareUnicodeCharacter{03B3}{\ensuremath{\gamma}}
\DeclareUnicodeCharacter{03BC}{\ensuremath{\mu}}
\DeclareUnicodeCharacter{03C3}{\ensuremath{\sigma}}
\DeclareUnicodeCharacter{2192}{\ensuremath{\rightarrow}}
\DeclareUnicodeCharacter{2264}{\ensuremath{\le}}
\DeclareUnicodeCharacter{2265}{\ensuremath{\ge}}
\DeclareUnicodeCharacter{2248}{\ensuremath{\approx}}
\DeclareUnicodeCharacter{2208}{\ensuremath{\in}}
\DeclareUnicodeCharacter{00D7}{\ensuremath{\times}}
\DeclareUnicodeCharacter{2605}{\textasteriskcentered}
\DeclareUnicodeCharacter{00A7}{\S}
\DeclareUnicodeCharacter{00B1}{\ensuremath{\pm}}
\DeclareUnicodeCharacter{00B7}{\textperiodcentered}
\DeclareUnicodeCharacter{2013}{--}
\DeclareUnicodeCharacter{2014}{---}
\DeclareUnicodeCharacter{2212}{\ensuremath{-}}
% Box-drawing for architecture diagram (code-fenced; renders as ASCII).
\DeclareUnicodeCharacter{2500}{-}
\DeclareUnicodeCharacter{2502}{|}
\DeclareUnicodeCharacter{251C}{+}
\DeclareUnicodeCharacter{2514}{+}
\DeclareUnicodeCharacter{2510}{+}
\DeclareUnicodeCharacter{2518}{+}
\DeclareUnicodeCharacter{250C}{+}
\DeclareUnicodeCharacter{25BC}{v}

% Pandoc compatibility (its default template defines these — we need
% them too when using a custom template, otherwise tightlist etc. fail)
\providecommand{\tightlist}{\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}
\providecommand{\real}[1]{#1}
\providecommand{\toprule}{\hline}
\providecommand{\midrule}{\hline}
\providecommand{\bottomrule}{\hline}

% Pandoc 3.x emits \refstepcounter{none} inside longtable cells when no
% caption is supplied — without a defined "none" counter this triggers
% a hyperref crash. Defining it harmlessly satisfies the reference.
\newcounter{none}
\providecommand{\thenone}{}
\providecommand{\theHnone}{none.\arabic{none}}

% Pandoc highlighting macros (in case any code blocks slip through with syntax)
$if(highlighting-macros)$
$highlighting-macros$
$endif$

% Pandoc may also emit \pandocbounded for images; provide a no-op.
\providecommand{\pandocbounded}[1]{#1}

%% ===========================================================================
%% TITLE + ABSTRACT — span both columns via \twocolumn[...] argument
%% ===========================================================================
% In a two-column document the title and abstract usually run across
% the full text width (canonical IEEE/ACM layout). \twocolumn[...]
% accepts content to typeset full-width before column flow starts.

\begin{document}

\twocolumn[
\begin{@twocolumnfalse}
% --- Title block ---
{\centering
\rule{\textwidth}{0.4pt}\\[0.3em]
{\Large\bfseries $title$\par}%
$if(subtitle)$
\vskip 0.2em
{\large $subtitle$\par}%
$endif$
\vskip 0.3em
\rule{\textwidth}{0.4pt}\\[1.5em]

% Authors — pandoc emits as a plain paragraph; we just center it.
$if(author)$
{\normalsize
$for(author)$$author$$sep$ \quad $endfor$
\par}
\vskip 0.3em
$endif$

$if(date)${\small\itshape $date$\par}\vskip 1em$endif$
\par}

% --- Abstract block (centered "Abstract" heading, indented body) ---
$if(abstract)$
\begin{quote}
\begin{center}{\bfseries Abstract}\end{center}
\vskip 0.3em
\noindent\small $abstract$
\end{quote}
\vskip 1em
$endif$
\end{@twocolumnfalse}
]

% --- Body content (flows in two columns) ---
$body$

\end{document}
